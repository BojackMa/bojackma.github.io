//始终有个区间是有序的，只和有序的区间比较就行了！！！这个二分的核心根本思想是去掉一半不可能的区间，这个是关键！！

class Solution {    
public:
    int search(vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            // 找到了
            if (nums[mid] == target) return mid;
            // 左半边有序
            if (nums[left] <= nums[mid]) {      //只和单调的区间去比较，另一个区间不去考虑
                // target 在左半边
                if (nums[left] <= target && target < nums[mid]) {
                    right = mid - 1;
                } else { // 否则去右半边
                    left = mid + 1;
                }
            }
            // 否则右半边有序
            else {
                // target 在右半边
                if (nums[mid] < target && target <= nums[right]) {
                    left = mid + 1;
                } else { // 否则去左半边
                    right = mid - 1;
                }
            }
        }
        return -1;
    }
};
