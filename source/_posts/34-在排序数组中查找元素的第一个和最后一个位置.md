class Solution {
    int findLeft(vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1;
        int ans = -1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target){
                ans = mid;     //找到了一个，记录下来
                right = mid - 1;            //此时不返回，继续找，所以说区间向左！！！！
            } 
            else if (nums[mid] >= target) { // 可能是答案，但还要往左看
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        return ans;
    }

    int findRight(vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1;
        int ans = -1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target){
                ans = mid;     //找到了一个，更新一下
                left = mid + 1;            //此时不返回，继续找，所以说区间向右！！！！
            } 
            else if (nums[mid] < target) { 
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return ans;
    }

public:
    vector<int> searchRange(vector<int>& nums, int target) {
        vector<int> ans;
        ans.push_back(findLeft(nums, target));
        ans.push_back(findRight(nums, target));
        return ans;
    }
};

//最终的解法就是在模版上进行微调，对left和right进行修改。

//这里的二分查找还不一样！！！因为不是非严格递增的序列，可能有多个mid符合要求，这样可能会提前结束。

//难不成用两遍二分查找
//我先找target和target + 1，从而解决问题，还是不要这样。找到了先记录一下，然后继续找，知道压迫到极点。
