关键的hashmap函数
```
unordered_map<KeyType, ValueType> mp;
//key找value，所以key一般是数字，value一般是下标
//1、插入操作
mp[key] = value;
//2、判断是否存在，返回1是存在，0是不存在
mp.count(key)
//3、取值
mp[key]

```
#include <iostream>
#include <vector>
#include <unordered_map>

using namespace std;

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> ump;
        vector<int> ans;

        for(int i = 0; i < nums.size(); i++){

            // 查找是否存在 target - nums[i]
            if(ump.count(target - nums[i]) == 1){
                ans.push_back(ump[target - nums[i]]);
                ans.push_back(i);
                return ans;
            }

            // 存入当前数字
            ump[nums[i]] = i;
        }

        return ans;
    }
};

int main() {

    int n;
    cin >> n;

    vector<int> nums(n);

    for(int i = 0; i < n; i++){
        cin >> nums[i];
    }

    int target;
    cin >> target;

    Solution s;
    vector<int> ans = s.twoSum(nums, target);

    for(int x : ans){
        cout << x << " ";
    }

    return 0;
}
