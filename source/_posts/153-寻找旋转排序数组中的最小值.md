```
//经典的思路：选择二分，总有一个区间是递增的，那个区间就比较好取最小值，然后取处理另一个区间
//不一定一定要找最小值，但是可以把所有的可能的最小值找到！！！
class Solution {
public:
    int findMin(vector<int>& nums) {
        int left = 0; int right = nums.size() - 1;
        int ans = 5005;
        while(left <= right){
            int mid = left + (right - left) / 2;
            //说明前面是有序区间
            if(nums[mid] >= nums[left]){
                ans = min(ans, nums[left]);
                left = mid + 1;
            }
            else{
                ans = min(ans, nums[mid]);
                right = mid - 1;
            }
        }
        return ans;
    }
};
```
