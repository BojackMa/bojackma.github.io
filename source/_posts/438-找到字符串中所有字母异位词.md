class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        multiset<char> standard;
        multiset<char> compare;

        for(auto pp : p)    standard.insert(pp);

        int left = 0;
        int right = p.size() - 1;

        vector<int> ans;
        if(s.size() < p.size()) return ans;
        for(int i = 0; i < p.size(); i++)  {
            compare.insert(s[i]);
        } 

        while(1){
            if(compare == standard) {
                ans.emplace_back(left);
            }
            compare.erase(compare.find(s[left]));
            left++;
            right++;
            if(right == s.size())   break;
            compare.insert(s[right]);
        }
        return ans;
    }
};
