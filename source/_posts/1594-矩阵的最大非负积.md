class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();     //还是要这么写，后面记得加括号
        vector<vector<long long int>> cursum_max(m, vector<long long int>(n, 0));    //这个表示能取到的最大值
        vector<vector<long long int>> cursum_min(m, vector<long long int>(n, 0));    //这个表示能取到的最小值

        cursum_max[0][0] = cursum_min[0][0] = grid[0][0];
        

        for(int i = 1; i < grid[0].size(); i++){
            cursum_max[0][i] = cursum_min[0][i] = cursum_max[0][i - 1] * grid[0][i];
        }


        for(int i = 1; i < grid.size(); i++){
            cursum_max[i][0] = cursum_min[i][0] = cursum_max[i - 1][0] * grid[i][0];
        }


        for(int i = 1; i < grid.size(); i++){     
            for(int j = 1; j < grid[0].size(); j++){
                
                long long int tmp1 = grid[i][j] *  cursum_max[i - 1][j];
                long long int tmp2 = grid[i][j] *  cursum_min[i - 1][j];
                
                long long int tmp3 = grid[i][j] *  cursum_max[i][j - 1];
                long long int tmp4 = grid[i][j] *  cursum_min[i][j - 1];

                cursum_max[i][j] = max(max(tmp1, tmp2), max(tmp3, tmp4));
                cursum_min[i][j] = min(min(tmp1, tmp2), min(tmp3, tmp4));

            }
        }

        if(cursum_max[m - 1][n - 1] >= 0)
            return cursum_max[m - 1][n - 1] % (1000000000 + 7);
        else return -1;
    }
};


//还是要选择最大值和最小值作为中间态，而不是最大正数或者最小负数，因为每个格子只有两种选择，上面或者左边，保存好这两种状态就可以

// 在 DP 中，如果一个状态未来可能“翻转价值”（比如乘负数），那你必须保留“极值”，而不是“类别”。保留极值（最有潜力变优的状态）！！！

//移动过程中，整体的绝对值在变化（总是变大，除非遇到0），因此我们需要保持最大值和最小值，而不是正数和负数
