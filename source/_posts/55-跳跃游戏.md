```
class Solution {
public:
    bool canJump(vector<int>& nums) {

        int curFar = 0;

        for(int i = 0; i < nums.size(); i++){

            if(i > curFar){     //说明这个点到不了
                return 0;
            }
            curFar = max(curFar, i + nums[i]);

        }

        return 1;       //遍历完了，说明都能到
    }
};
```
