class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> ans;  //这个是默认为空！！！是完全合法且已初始化的对象，它表示创建一个 空的 vector<vector<int>>.
        ans.push_back({1});
        for(int i = 2; i <= numRows; i++){
            vector<int> curRow(i);
            curRow[0] = 1; curRow[i - 1] = 1;
            for(int j = 1; j < i - 1; j++){     //对于第i行第j列的元素来说
                curRow[j] = ans[i - 2][j - 1] + ans[i - 2][j];  
            }
            ans.push_back(curRow);
        }
        return ans;
    }
};
