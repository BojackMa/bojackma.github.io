class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        //先确定在哪一行，再确定在哪一列。
        int m = matrix.size();
        int n = matrix[0].size();
        int leftHang = 0;
        int rightHang = m - 1;
        int midHang;

        while(leftHang <= rightHang){
            midHang = leftHang + (rightHang - leftHang) / 2;
            if(matrix[midHang][0] == target)    return 1;
            else if(matrix[midHang][0] > target) rightHang = midHang - 1;
            else  leftHang = midHang + 1;
        }

        //此时leftHang停留在第一个大于target的位置  
        int targetHang = leftHang - 1;
        if(targetHang < 0)  return 0;      

        int leftLie = 0;
        int rightLie = n - 1;
        int midLie;

        while(leftLie <= rightLie){
            midLie = leftLie + (rightLie - leftLie) / 2;
            if(matrix[targetHang][midLie] == target)    return 1;
            else if(matrix[targetHang][midLie] > target) rightLie = midLie - 1;
            else  leftLie = midLie + 1;
        }

        //走到这还没有找到，那就是没有了
        return 0;

    }
};
