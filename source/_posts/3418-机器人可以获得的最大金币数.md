class Solution {
public:
    int maximumAmount(vector<vector<int>>& coins) {
        int m = coins.size(), n = coins[0].size();
        const int INF = -1e9;
        vector dp(m, vector(n, vector<int>(3, INF)));

        // 初始化
        if (coins[0][0] >= 0) {
            dp[0][0][0] = coins[0][0];
        } else {
            dp[0][0][0] = coins[0][0]; // 不感化
            dp[0][0][1] = 0;           // 感化
        }

        //两种情况：格子为正，dp[i][j][k] = max(上, 左) + coins[i][j]，不要用感化次数
        //格子为负，不感化，dp[i][j][k] = max(上, 左) + coins[i][j]；感化，dp[i][j][k] = max(上, 左，且来自 k-1) + 0


        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k <= 2; k++) {
                    if (i == 0 && j == 0) continue;
                    int val = coins[i][j];
                    // 从上 or 左
                    if (i > 0) {
                        // 不感化
                        dp[i][j][k] = max(dp[i][j][k], dp[i-1][j][k] + val);

                        // 感化
                        if (val < 0 && k > 0) {
                            dp[i][j][k] = max(dp[i][j][k], dp[i-1][j][k-1]);
                        }
                    }

                    if (j > 0) {
                        // 不感化
                        dp[i][j][k] = max(dp[i][j][k], dp[i][j-1][k] + val);

                        // 感化
                        if (val < 0 && k > 0) {
                            dp[i][j][k] = max(dp[i][j][k], dp[i][j-1][k-1]);
                        }
                    }
                }
            }
        }

        return max({dp[m-1][n-1][0],
                    dp[m-1][n-1][1],
                    dp[m-1][n-1][2]});

    }

};
