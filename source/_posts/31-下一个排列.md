下面这两个可能比较常用
```
swap(nums[i], nums[index]); 
reverse(nums.begin() + j, nums.end());
```

```
class Solution {
public:
    void nextPermutation(vector<int>& nums) {
        int n = nums.size();
        int j = n - 1;

        // 从右向左找到第一个不满足降序的点，因为排列的变化只可能从尾部开始最小调整。
        while (j > 0 && nums[j] <= nums[j - 1]) {
            j--;
        }

        // 如果j == 0，说明整个数组是降序，直接反转，这是特殊情况
        if (j == 0) {
            reverse(nums.begin(), nums.end());
            return;
        }

        int i = j - 1; // 记录第一个不满足降序的位置，这里是元素5
        int index = n - 1;

        // 从右向左找到第一个比nums[i]大的元素，这里是元素6
        while (nums[index] <= nums[i]) {
            index--;
        }

        // 交换
        swap(nums[i], nums[index]);    //5764变成6754

        // 反转j到末尾的部分，使其变为升序
        reverse(nums.begin() + j, nums.end());    //6754变成6457
    }
};
```
