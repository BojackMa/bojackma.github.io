```
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {

        unordered_set<int> us;
        int ans = 0;

        for(int i = 0; i < nums.size(); i++){
            us.insert(nums[i]);
        }

        //选择遍历容器而不是数组是因为数组里面重复的太多了，us可以去重
        for (auto it = us.begin(); it != us.end(); ++it) {  //所有 STL 容器都可以用迭代器这种方式遍历
            int nums = *it;
            if(us.count(nums - 1) == 1) continue;    //我们要抓到数字的头
            int tail = nums;
            while(us.count(tail) == 1){         //找到尾巴
                tail++;
            }
            ans = max(ans, tail - nums);

        }

        return ans;

    }
};
//时间复杂度应该是o（3n）
```
