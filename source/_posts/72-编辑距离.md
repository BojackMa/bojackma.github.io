class Solution {
public:
    int minDistance(string word1, string word2) {
        int m = word1.size();
        int n = word2.size();

        // dp[i][j]：word1 前 i 个字符 转换成 word2 前 j 个字符的最少操作数
        vector<vector<int>> dp(m + 1, vector<int>(n + 1));

        // 初始化
        for (int i = 0; i <= m; ++i) dp[i][0] = i;  // 删除
        for (int j = 0; j <= n; ++j) dp[0][j] = j;  // 插入

        for (int i = 1; i <= m; ++i) {
            for (int j = 1; j <= n; ++j) {

                if (word1[i - 1] == word2[j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1];  // 字符相同
                } 
                else {
                    int replace = dp[i - 1][j - 1];
                    int del = dp[i - 1][j];
                    int insert = dp[i][j - 1];

                    dp[i][j] = min({replace, del, insert}) + 1;
                }
            }
        }

        return dp[m][n];
    }
};


// 本题可以使用动态规划解决。
// 定义 dp[i][j] 表示 word1 的前 i 个字符转换成 word2 的前 j 个字符所需要的最少操作数。
// 当 word1[i-1] == word2[j-1] 时，两个字符相同，不需要额外操作，因此 dp[i][j] = dp[i-1][j-1]。
// 当字符不同时，可以进行三种操作：替换、删除、插入，分别对应
// dp[i-1][j-1] + 1、dp[i-1][j] + 1、dp[i][j-1] + 1，取三者的最小值即可。
// 初始状态下，当一个字符串为空时，只能通过插入或删除操作进行转换，因此 dp[i][0] = i，dp[0][j] = j。
// 最终答案为 dp[m][n]。
