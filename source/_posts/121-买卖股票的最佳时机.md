class Solution {
public:
    int maxProfit(vector<int>& prices) {
        
        int curmin = prices[0];
        int ans = 0;

        for(auto price : prices){
            
            //如果今天涨了，尝试卖出
            if(price >= curmin) {
                ans = max(ans, price - curmin);
                continue;
            }

            //如果今天价格比之前的都要低，今天买入
            if(price < curmin) {
                curmin = price;
            }

        }
        
        return ans;

    }
};



//我一开始的思路比较蠢，先找到数组的开始增长的那个点，因为前面下降的点肯定不是入手的点。
// 找到开始增长的点后，记录为最小值，往后找到最大值，相减得到结果。
// 然后再从最大值后面找比最小值还小的值，再往后找最大值，相减得到结果。如此往复。本质上是在找“每个低谷-高峰对”，这个复杂度on2。
