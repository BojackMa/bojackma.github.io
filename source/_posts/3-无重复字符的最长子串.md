class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> uset;
        if(s.length() == 0) return 0;
        int left = 0;
        int right = 1;
        int ans = 1;
        uset.insert(s[0]);
        while(right < s.size()){
            //如果set中有这个字符，那就左边窗口向右移动
            if(uset.count(s[right]) == 1){
                while(s[left] != s[right]){
                    uset.erase(s[left]);   //去除这些字母
                    left ++;
                }
                left ++;        //要画个图，还要跳过这个字符
            }
            //如果没有，那就加入这个字符，同时更新ans
            else{
                uset.insert(s[right]);
                ans = max(ans, right - left + 1);
            }
            right++;
        }
        return ans;

    }
};
