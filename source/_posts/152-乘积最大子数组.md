```
这道题和 第300题 相似，DP 数组的定义要想一想。DP 数组并不是直接存储题目的最终答案，而是更关注状态的可传递性，即能否将问题拆解为子问题。因此，我们需要以 nums[i] 结尾来定义 DP 数组。
```


```
class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int res = nums[0];  //先给一个值
        vector<int> minDp(nums.size());
        vector<int> maxDp(nums.size());

        //初始化dp数组
        for(int i = 0; i < nums.size(); i++){
            minDp[i] = nums[i];
            maxDp[i] = nums[i];
        }

        //开始更新
        for(int i = 1; i < nums.size(); i++){
            int tmp1 = nums[i] * minDp[i - 1];
            int tmp2 = nums[i] * maxDp[i - 1];
            minDp[i] = min(min(tmp1, tmp2), minDp[i]);  //C++ 的 min / max 并不支持三个参数，要写成这样。
            maxDp[i] = max(max(tmp1, tmp2), maxDp[i]);
            res = max(res, maxDp[i]);
        }

        return res;

    }
};
//minDp表示以 nums[i] 结尾的 最小乘积
//maxDp以 nums[i] 结尾的 最大乘积   

//这个我当时想不到怎么处理负数，其实很简单的，再开个数组保存一下，也就是保存最小负数和最大正数，这样就行了

//不管怎么说，加上前面的数组，两个数组肯定有所变化
```
