#include <iostream>
#include <vector>
#include <string>
#include <unordered_map>
#include <algorithm>

using namespace std;

class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        
        unordered_map<string, vector<string>> umap;
        vector<vector<string>> ans;

        for(int i = 0; i < strs.size(); i++){
            string curstr = strs[i];
            sort(curstr.begin(), curstr.end()); //得到正序，然后以正序插入str
            umap[curstr].push_back(strs[i]);
        }
        
        //然后遍历hash表
        for(auto it = umap.begin(); it != umap.end(); it++){
            ans.push_back(it -> second);
        }

        return ans;
    }
};

int main() {
    int n;
    cin >> n;  // 输入字符串个数
    vector<string> strs(n);
    
    for(int i = 0; i < n; i++){
        cin >> strs[i];  // 输入每个字符串
    }

    Solution sol;
    vector<vector<string>> res = sol.groupAnagrams(strs);

    // 输出结果，每个分组占一行，字符串用空格分隔
    for(auto &group : res){
        for(auto &s : group){
            cout << s << " ";
        }
        cout << endl;
    }
    return 0;
}

//核心思想：
//将相同字典序的字符串放到一起

//可以用it -> second来取数字
