```
class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n = nums.size();
        int ans = 0;
        vector<int> dp(n, 1);  // dp[i] 表示以 nums[i] 结尾的最长上升子序列长度

        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < i; ++j) {
                if (nums[j] < nums[i]) {
                    dp[i] = max(dp[i], dp[j] + 1);
                }
            }
        }

        for (auto num : dp) 
            if(num > ans)   ans = num;
    
        return ans;  // 返回 dp 中的最大值
    }
};

//时间复杂度最少的是贪心+二分查找，这里我没有看。
```
