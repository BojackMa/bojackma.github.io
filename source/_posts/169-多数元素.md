```
class Solution {
public:
    int majorityElement(vector<int>& nums) {
        unordered_map<int, int> umap;
        int count = 0; 
        int winner = 0;

        for(int i = 0; i < nums.size(); i++){
            umap[nums[i]]++;
            if(umap[nums[i]] > count){
                count = umap[nums[i]];
                winner = nums[i];
            }
        }

        return winner;

    }
};

//还是用hash表比较好吧，记录看过的数字
```
