```
如何判断是否要用DP？这个很关键！！！我觉得一定要记住这个！！！
1 是否要求“最值 / 方案数 / 是否可以”
2 是否可以拆成更小的子问题
3 子问题是否会重复
```

```
class Solution {
public:
    int rob(vector<int>& nums) {

        int n = nums.size();
        if (n == 1) return nums[0];
        if (n == 2) return max(nums[0], nums[1]);

        // dp[i] 表示：前 i 间房子能偷到的最大金额
        vector<int> dp(n + 1);

        dp[0] = 0;
        dp[1] = nums[0];
        dp[2] = max(nums[0], nums[1]);

        for (int i = 3; i <= n; i++) {
            // 不偷第 i 间 或 偷第 i 间
            dp[i] = max(dp[i - 1], dp[i - 2] + nums[i - 1]);
        }

        return dp[n];
    }
};
```
