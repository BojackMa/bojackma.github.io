class Solution {
public:
    vector<int> partitionLabels(string s) {

        vector<int> ans;

        unordered_map<char, int> umap;
        for(int i = 0; i < s.size(); i++){
            umap[s[i]] = i;
        }//记录每个字符的最后一个位置

        int cutPos = 0;
        int begincutPos = 0;
        for(int i = 0; i < s.size(); i++){
            //更新结束的切割位置
            cutPos = max(cutPos, umap[s[i]]);

            //开始切割，记录开始的切割位置
            if(i == cutPos){
                ans.push_back(cutPos - begincutPos + 1);
                begincutPos = i + 1;
            }
        }

        return ans;

    }
};
