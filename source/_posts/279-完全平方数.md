```
class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n + 1, INT_MAX); //或者用n + 1来代表最大值
        dp[0] = 1;
        dp[1] = 1;
        for(int i = 2; i <= n; i++){
            int squareroot = sqrt(i); //在<cmath>库里面，double类型
            if(i == squareroot * squareroot)    dp[i] = 1;
            for(int j = 1; j <= squareroot; j++)
                dp[i] = min(dp[i], dp[i - j * j] + 1);
        }
        return dp[n];
    }
};
```
