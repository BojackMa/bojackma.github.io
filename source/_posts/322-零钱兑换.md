class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        vector<int> dp(amount + 1, -1);

        //初始化
        dp[0] = 0;
        for(auto coin: coins)
            if(coin < amount + 1)   dp[coin] = 1;

        //一般情况
        for(int i = 1; i <= amount; i++){
            for(auto coin: coins){
                if(i >= coin && dp[i - coin] != -1){
                    if(dp[i] == -1) dp[i] = dp[i - coin] + 1;
                    else dp[i] = min(dp[i], dp[i - coin] + 1);
                }   
            }
        }

        return dp[amount];
    }
};
