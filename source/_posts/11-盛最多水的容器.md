```
class Solution {
public:
    int maxArea(vector<int>& height) {
        int begin = 0;
        int end = height.size() - 1;
        int ans = (end - begin) * min(height[end], height[begin]);
        while(begin < end){
            if(height[end] < height[begin]) end--;
            else begin++;
            ans = max(ans, (end - begin) * min(height[end], height[begin]));
        }
        return ans;
    }
};


// area = (j - i) * min(height[i], height[j])，因此当向里移动指针的时候，只有两个挡板都变高才可以，所以总是移动最短的那个指针，如果有更优解，那么挡板高度都会变高，本质上是利用单调性，系统性剪枝。
```
