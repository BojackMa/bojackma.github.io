```
class Solution {
public:
    int longestValidParentheses(string s) {
        int maxans = 0;
        int start = 0;       // 当前有效子串的起始位置
        stack<int> stk;      // 只存未匹配的 '(' 的下标

        for (int i = 0; i < s.length(); i++) {
            if (s[i] == '(') {
                stk.push(i);
            } 
            else if (s[i] == ')')  {
                if (stk.empty())     
                    start = i + 1;      // ')' 找不到匹配，当前有效串断掉，重置起点
                else {
                    stk.pop();  // 匹配掉一个 '('
                    if (stk.empty()) {
                        // 栈空：从 start 到 i 全部有效
                        maxans = max(maxans, i - start + 1);
                    } else {
                        // 栈非空：从栈顶的 '(' 后一位到 i 有效
                        maxans = max(maxans, i - stk.top());
                    }
                }
            }
        }
        return maxans;
    }
};


//看到'('、')'这个就应该想到用栈，栈里面存储index，方便计算！！！

//栈里面只保存'('，用来记录需要匹配的
//如果栈里面为空，又遇到了来了个')'，那说明这个')'永远也匹配不上了，要清掉了。
//栈里只放未匹配的 ( 的下标，边界信息单独用 start 记录。
```
