```
class Solution {
public:
    int jump(vector<int>& nums) {

        int curFar = 0;          // 当前跳能到的最远边界
        int nextFar = 0;         // 下次跳能到的最远位置
        int curJumptime = 0;     // 已跳跃次数

        if(nums.size() == 1) return 0;  // 特判，原地就是终点

        for(int i = 0; i < nums.size() - 1; i++){ // 遍历到倒数第二个元素就够了
            nextFar = max(nextFar, i + nums[i]); // 更新当前跳范围内能到的最远位置

            if(i == curFar){    // 到达当前跳的边界，必须跳一次
                curJumptime++;
                curFar = nextFar;
            }
        }

        return curJumptime;
    }
};
```
