```
//2、像“分割等和子集”这样的题目与序列无关，更关注“和”，DP 数组通常以和为状态，也就是典型的 0-1 背包问题。拆开来看，状态转移就是：选还是不选 nums[i]，能否凑出特定的和。

//这个问题等价于：能否从数组中选出一些元素，它们的和恰好是 sum(nums)/2？
//这就是经典的“0-1 背包”问题：每个数字只能选一次，目标容量是 sum/2。
//定义 dp[i][j] 表示：前 i 个数字，是否可以组成和 j

//先画一画，先用二维
```

```
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int sum = 0;
        int n = nums.size();
        for(auto num : nums)
            sum = sum + num;
        if(sum % 2 == 1)    return 0;
        int target = sum / 2;

        vector<vector<int>> dp(n + 1, vector<int>(target + 1));       //定义 dp[i][j] 表示：前 i 个数字，是否可以组成和 j
        for (int i = 0; i <= n; i++) dp[i][0] = true;  // 前 i 个数字都可以凑出 0

        // 状态转移
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= target; j++) {

                if (j >= nums[i - 1]) {
                    dp[i][j] = max(dp[i-1][j], dp[i-1][j - nums[i - 1]]);       //状态转移方程
                } else {
                    dp[i][j] = dp[i-1][j];
                }

            }
        }

        return dp[n][target];
    }
};
```
