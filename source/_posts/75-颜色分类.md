class Solution {
public:
    void sortColors(vector<int>& nums) {
        int index = 0;        //这个标志着下一个待处理的数字，目前是
        while(index < nums.size() && nums[index] == 0){
            index++;
        }                       //index初始化完毕
   
        for(int i = index; i < nums.size(); i++){       //i标志着下一个为0，且需要交换的
            if(nums[i] == 0){
                swap(nums[i], nums[index]);
                index++;
            }       //如果不为0，说明是目标用户
        }            
        
        while(index < nums.size() && nums[index] == 1){
            index++;
        }                       //index初始化完毕


        for(int i = index; i < nums.size(); i++){       //i标志着下一个为1，且需要交换的
            if(nums[i] == 1){
                swap(nums[i], nums[index]);
                index++;
            }       //如果不为0，说明是目标用户
        }       
    }
};


//一般原地操作都会用到指针。
