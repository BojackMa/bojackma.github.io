class Solution {
public:
    string longestPalindrome(string s) {
        
        int winner = 0;
        int winnerLeft, winnerRight;

        for(int i = 0; i < s.size(); i++){
            int left = i;
            int right = i;
            while(left >= 0 && right < s.size()){
                if(s[left] == s[right]){
                    if(winner < right - left + 1){
                        winner = max(winner, right - left + 1);
                        winnerLeft = left;
                        winnerRight = right;
                    }
                    left--;
                    right++;
                }
                else{
                    break;
                }
            }
        }

        for(int i = 0; i < s.size(); i++){
            int left = i;
            int right = i + 1;
            while(left >= 0 && right < s.size()){
                if(s[left] == s[right]){
                    if(winner < right - left + 1){
                        winner = max(winner, right - left + 1);
                        winnerLeft = left;
                        winnerRight = right;
                    }
                    left--;
                    right++;
                }
                else{
                    break;
                }
            }
        }

        return s.substr(winnerLeft, winnerRight - winnerLeft + 1);

    }
};
